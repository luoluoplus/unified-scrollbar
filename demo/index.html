💻 Usage
// Install: npm install unified-scrollbar import UnifiedScrollbar from 'unified-scrollbar'; // ① Apply default theme globally (one line!) UnifiedScrollbar.apply(); // ② Use a preset theme UnifiedScrollbar.applyTheme('dark'); UnifiedScrollbar.applyTheme('neon'); // ③ Fully custom UnifiedScrollbar.apply({ width: '10px', thumbColor: '#7c6dfa', thumbHoverColor: '#a78bfa', trackColor: 'rgba(0,0,0,0.1)', borderRadius: '5px', }); // ④ Target a specific element UnifiedScrollbar.apply({ target: '#sidebar' }); // ⑤ Remove styles UnifiedScrollbar.remove();
const UnifiedScrollbar = require('unified-scrollbar'); UnifiedScrollbar.apply(); UnifiedScrollbar.applyTheme('macos');
<!-- Via CDN (jsDelivr) --> <script src="https://cdn.jsdelivr.net/npm/unified-scrollbar/dist/index.umd.js"></script> <script> UnifiedScrollbar.apply(); // or UnifiedScrollbar.applyTheme('soft'); </script>
/* Option A: Import in CSS */ @import 'unified-scrollbar/css'; /* Option B: Import in JS/TS */ import 'unified-scrollbar/style.css';
// Next.js — _app.tsx import { useEffect } from 'react'; import type { AppProps } from 'next/app'; export default function App({ Component, pageProps }: AppProps) { useEffect(() => { import('unified-scrollbar').then(m => m.default.applyTheme('dark')); }, []); return <Component {...pageProps} />; } // SSR CSS generation (server-side) import { generateCSS } from 'unified-scrollbar'; const css = generateCSS({ thumbColor: '#7c6dfa' });
